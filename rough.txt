<script>
  function setupScrollSlide(parentId, type) {
    const parent = document.getElementById(parentId);
    const img1 = parent.querySelector('.img1');
    const img2 = parent.querySelector('.img2');
    const img3 = parent.querySelector('.img3');

    function update() {
      const rect = parent.getBoundingClientRect();
      const vh = window.innerHeight;

      // only animate when in view
      if (rect.bottom < 0 || rect.top > vh) return;

      // progress: 0 when top just enters bottom of viewport, 1 when completely passed
      let progress = (vh - rect.top) / (vh + rect.height);
      progress = Math.min(1, Math.max(0, progress));

      const translate = progress * 100; // percent

      if (type === 'p1') {
        img1.style.transform = `translateX(-${translate}%)`;
        img2.style.transform = `translateX(-${translate}%)`;
      } else if (type === 'p2') {
        img2.style.transform = `translateX(${translate}%)`;
        img3.style.transform = `translateX(${translate}%)`;
      }
    }

    window.addEventListener('scroll', update);
    window.addEventListener('resize', update);
    update();
  }

  setupScrollSlide('img-animation-parent-1', 'p1');
  setupScrollSlide('img-animation-parent-2', 'p2');
</script>

  <script>
    // Animation observer for both parents
    function setupObserver(parentId, className, reverseMs = 3000) {
      const parent = document.getElementById(parentId);
      let animated = false;
      const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting && !animated) {
            animated = true;
            parent.classList.add('animate');
            setTimeout(() => {
              parent.classList.remove('animate');
              animated = false;
            }, reverseMs);
          }
        });
      }, { threshold: 1.0 });
      observer.observe(parent);
    }

    setupObserver('img-animation-parent-1', 'p1');
    setupObserver('img-animation-parent-2', 'p2');
  </script>
